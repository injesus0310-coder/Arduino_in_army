#include <LiquidCrystal.h>

LiquidCrystal lcd(12,11,5,4,3,2);

void setup()
{
  lcd.begin(16,2);
}

void loop()
{
  int value = analogRead(A0);
  float voltage = value*5.0/1023;
  float temper = voltage*100-50;
  
//  lcd.print("temper ");  
//  lcd.print(temper); 이런 방식도 가능하겠지만
  String tempStr = String(temper);
  lcd.print("Temp = "+tempStr);
  delay(500);
  lcd.clear();
}
